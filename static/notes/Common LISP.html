Define a procedure that takes three numbers as argumnest and returns the sum of the squares of the two larger numbers. 

(define (max-two a b c)
  (if (>= a b)
      (if (>= b c)
          (list a b)
          (list a c))
      (if (>= a c)
          (list b a)
          (list b c))))

(define (square x)
  (* x x))

(define (sum-of-squares x y)
  (+ (square x) (square y))

(define (sum-max a b c)
  (sum-of-squares (max-two a b c))
